1
Add in web.config in system.web section

        <system.web>
			<httpModules>
				<add name="UploadModule" type="Areas.Lib.UploadProgress.UploadModule, Areas.Lib" />
			</httpModules>		
		</system.web>

		<httpRuntime maxRequestLength="2097151" requestValidationMode="2.0" executionTimeout="999999" />

2
Add in web.config in system.webServer section

	<system.webServer>
		  <modules runAllManagedModulesForAllRequests="true">
		    <add name="UploadModule" type="Areas.Lib.UploadProgress.UploadModule, Areas.Lib" preCondition="managedHandler"/>
		  </modules>

		  <validation validateIntegratedModeConfiguration="false" ></validation>
			<security>
			  <requestFiltering>
				<requestLimits maxAllowedContentLength="2147482624"></requestLimits>
			  </requestFiltering>
			</security>
	<system.webServer>
