using Areas.Lib.DataBootstrap;
    using System;
    using System.Collections.Generic;
    using System.Linq;
    using System.Web;
    using System.Web.Mvc;

    public class BootstrapController : Controller
    {
        [HttpPost]
        public void UpdateColumn(string columnName, string text1000)
        {
            using (var db = new SampleDbHelper())
            {
                db.SetColumn(columnName, text1000);
            }
        }

        [HttpGet]
        public string GetTopColumn(string columnName)
        {
            using (var db = new SampleDbHelper())
            {
                return db.GetTopColumn(columnName).ToString();
            }
        }
    }
